# @longhouse/client

Generated by csilgen. A typed CSIL client: the generated codec owns CBOR
(de)serialization and the [`csilgen-transport`](https://www.npmjs.com/package/csilgen-transport)
library owns the envelope, framing, and connection lifecycle. You supply only a
*carrier* that moves bytes, so the same typed surface rides HTTP, TLS, a
WebSocket, WebRTC, QUIC, or raw UDP unchanged.

## Install

```sh
npm install @longhouse/client csilgen-transport
```

## CSIL-RPC (HTTP)

Request/response. The library owns the envelope (`RpcRequest`/`RpcResponse`);
you bring a carrier that moves bytes. The HTTP carrier below is just one
example — swap `fetch` for any client (it implements the generated byte seam).

```ts
import { RpcRequest, RpcResponse } from "csilgen-transport";
import { AsyncApiClient, type AsyncServiceTransport } from "@longhouse/client";

// One example carrier: CSIL-RPC over an HTTP POST. The library owns the envelope;
// the carrier owns only the transport. Swap `fetch` for any HTTP client.
class HttpRpcCarrier implements AsyncServiceTransport {
  constructor(private readonly baseUrl: string) {}

  async call(service: string, op: string, req: Uint8Array): Promise<Uint8Array> {
    const envelope = new RpcRequest(service, op, req).encode();
    const http = await fetch(this.baseUrl.replace(/\/+$/, "") + "/csil/v1/rpc", {
      method: "POST",
      headers: { "Content-Type": "application/cbor", Accept: "application/cbor" },
      // `Uint8Array` is a valid fetch body at runtime; the cast satisfies the DOM
      // `BodyInit` typing.
      body: envelope as BodyInit,
    });
    if (!http.ok) throw new Error(`csil-rpc ${service}/${op}: http ${http.status}`);

    // `intoTransportError` throws a `TransportError` for any non-zero transport status.
    const resp = RpcResponse.decode(
      new Uint8Array(await http.arrayBuffer()),
    ).intoTransportError();
    // A typed application error rides as a status-0 `ServiceError` variant — distinct
    // from a transport failure. Surface it so the typed client decodes success only.
    if (resp.variant === "ServiceError") {
      throw new Error(`csil-rpc ${service}/${op}: ServiceError`);
    }
    return resp.payload;
  }
}

async function main() {
  const client = new AsyncApiClient(new HttpRpcCarrier("http://localhost:5080"));
  const resp = await client.audit.queryAudit({ houseId: undefined as any });
  console.log(resp);
}

main();
```

## CSIL-Events (TLS)

Typed, bidirectional event streams over a long-lived connection. The library
owns the `$hello`/`$hello-ack` handshake, the `$ping`/`$pong` heartbeat, and
framing; the generated router dispatches typed events. The TLS carrier below is
just one example — a WebSocket/WebTransport/QUIC carrier drops in unchanged.

```ts
import {
  Event,
  Hello,
  HelloAck,
  Heartbeat,
  control,
  frameLengthPrefixed,
  LengthPrefixedDeframer,
  type FrameCarrier,
  type Profile,
} from "csilgen-transport";
import * as tls from "node:tls";

// One example carrier: a TLS byte stream framed with CSIL's 4-byte length prefix.
function openTlsCarrier(host: string, port: number): FrameCarrier {
  const socket = tls.connect({ host, port });
  const deframer = new LengthPrefixedDeframer();
  const inbox: Uint8Array[] = [];
  const waiters: ((f: Uint8Array | null) => void)[] = [];
  const deliver = (f: Uint8Array | null) => {
    const w = waiters.shift();
    if (w) w(f);
    else if (f) inbox.push(f);
  };
  socket.on("data", (chunk: Uint8Array) => {
    deframer.push(chunk);
    for (let f = deframer.next(); f !== null; f = deframer.next()) deliver(f);
  });
  socket.on("close", () => deliver(null));
  return {
    sendFrame(bytes: Uint8Array) {
      socket.write(frameLengthPrefixed(bytes));
    },
    recvFrame(): Promise<Uint8Array | null> {
      const f = inbox.shift();
      return f ? Promise.resolve(f) : new Promise((res) => waiters.push(res));
    },
  };
}


async function session() {
  const carrier = openTlsCarrier("localhost", 7443);

  // $hello / $hello-ack handshake (control plane).
  await carrier.sendFrame(new Hello([1], ["verbose"]).encode());
  const ackFrame = await carrier.recvFrame();
  if (ackFrame === null) throw new Error("connection closed during handshake");
  const profile = HelloAck.decode(ackFrame).profile as Profile;

  // Recv loop: answer $ping with $pong. This package declares no <->/<- operations,
  // so there is no generated channel router to dispatch typed events into.
  for (let frame = await carrier.recvFrame(); frame !== null; frame = await carrier.recvFrame()) {
    const ev = Event.decode(frame, profile);
    if (ev.event === control.PING_NAME) {
      const ping = Heartbeat.decode(ev.payload);
      await carrier.sendFrame(
        Event.verbose(undefined, control.PONG_NAME, new Heartbeat(ping.nonce).encode()).encode(profile),
      );
    }
  }
}

session();
```

## CSIL-Datagrams (UDP)

Unreliable, unordered, message-oriented. The library owns the `Datagram`
envelope; you bring a datagram carrier. The UDP carrier below is one example —
a WebRTC unreliable DataChannel or QUIC datagrams drop in unchanged.

```ts
import { Datagram, type DatagramCarrier } from "csilgen-transport";
import * as dgram from "node:dgram";
import { toAuditQueryCbor, fromAuditPageCbor } from "@longhouse/client";
import type { AuditQuery, AuditPage } from "@longhouse/client";

// One example carrier: UDP via node's dgram. Datagrams are unreliable and
// unordered, so the carrier never waits for or correlates a reply.
function openUdpCarrier(host: string, port: number): DatagramCarrier {
  const socket = dgram.createSocket("udp4");
  const inbox: Uint8Array[] = [];
  const waiters: ((d: Uint8Array | null) => void)[] = [];
  socket.on("message", (msg: Uint8Array) => {
    const w = waiters.shift();
    if (w) w(msg);
    else inbox.push(msg);
  });
  return {
    sendDatagram(bytes: Uint8Array) {
      socket.send(bytes, port, host);
    },
    recvDatagram(): Promise<Uint8Array | null> {
      const d = inbox.shift();
      return d ? Promise.resolve(d) : new Promise((res) => waiters.push(res));
    },
  };
}

// The operation's datagram ordinal — its @wire-id, or a channel-agreed number.
const OP_ORD = 1;

async function main() {
  const carrier = openUdpCarrier("localhost", 9000);

  // Fire-and-forget: encode the `->` request and send it. seq 0 marks an
  // unsequenced datagram.
  const req: AuditQuery = { houseId: undefined as any };
  await carrier.sendDatagram(new Datagram(OP_ORD, 0, toAuditQueryCbor(req)).encode());

  // Recv path: a datagram of the RESPONSE type MAY arrive later — or never. There is
  // NO synchronous response; the caller must tolerate loss and reordering and handle
  // a reply whenever (if ever) it shows up.
  const inbound = await carrier.recvDatagram();
  if (inbound !== null) {
    const dg = Datagram.decode(inbound);
    const resp: AuditPage = fromAuditPageCbor(dg.payload);
    console.log("late response", resp);
  }
}

main();
```
